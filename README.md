```

|- lib
|- web
|- test
```

```
  |- commom/
  |- config/
    |- routes/
    |- themes/
  |- constrants/
  |- dialogs/
  |- localization/
  |- models/
  |- modules/
  |- repositories/
  |- snackbar/
  |- utils/
    |- helpers/
    |- services/
  |- widgets/
|- main.dart

```
